# Phase 3 VMM Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Relocate the Aegis kernel to higher-half virtual address `0xFFFFFFFF80000000`, establish permanent managed page tables, and expose `vmm_map_page` / `vmm_unmap_page`.

**Architecture:** `boot.asm` builds identity + higher-half page tables before calling `kernel_main`; the linker splits VMA (`0xFFFFFFFF80000000`) from LMA (`0x100000`) for all C sections; `vmm_init()` replaces boot tables with PMM-allocated ones and switches `cr3`.

**Tech Stack:** x86-64 NASM, C freestanding (`-mcmodel=kernel`), GNU ld, GRUB/multiboot2, QEMU

```ld
OUTPUT_ARCH(i386:x86-64)

PHYS_BASE = 0x100000;
KERN_VMA  = 0xFFFFFFFF80000000;

SECTIONS
{
    . = PHYS_BASE;

    /* Boot trampoline: VMA = LMA = physical.
     * 32-bit setup code, GDT, and early 64-bit trampoline run here before
     * jumping to the higher-half virtual address. */
    .multiboot : { KEEP(*(.multiboot)) }
    .text.boot  : { *(.text.boot) }

    /* Switch to higher-half virtual addresses.
     * AT(...) sets the LMA (physical load address) for each section. */
    . += KERN_VMA;

    .text   : AT(ADDR(.text)   - KERN_VMA) { *(.text .text.*) }
    .rodata : AT(ADDR(.rodata) - KERN_VMA) { *(.rodata .rodata.*) }
    .data   : AT(ADDR(.data)   - KERN_VMA) { *(.data .data.*) }

    .bss : AT(ADDR(.bss) - KERN_VMA) {
        *(COMMON)
        *(.bss .bss.*)
    }
    _kernel_end = .;   /* higher-half virtual address */
}
```

```c
#ifndef VMM_H
#define VMM_H

#include <stdint.h>

/* Page-table entry flag bits (x86-64 PTE format). */
#define VMM_FLAG_PRESENT   (1UL << 0)
#define VMM_FLAG_WRITABLE  (1UL << 1)
#define VMM_FLAG_USER      (1UL << 2)
#define VMM_FLAG_NX        (1UL << 63)

/* Initialise the virtual memory manager.
 * Requires pmm_init() to have been called first.
 * Allocates permanent kernel page tables from the PMM, switches cr3,
 * and prints [VMM] OK. Panics on failure. */
void vmm_init(void);

/* Map one 4KB page: virt -> phys with given flags.
 * Allocates intermediate page tables from PMM as needed.
 * Panics if virt or phys is not PAGE_SIZE-aligned, or if virt is already mapped. */
void vmm_map_page(uint64_t virt, uint64_t phys, uint64_t flags);

/* Unmap one 4KB page. Panics if not mapped.
 * Does not free intermediate tables (deferred to a future reclaim pass). */
void vmm_unmap_page(uint64_t virt);

#endif /* VMM_H */
```

## Task 8: Create vmm.c

**Files:**
- Create: `kernel/mm/vmm.c`
- Modify: `Makefile` (add to MM_SRCS)

- [ ] **Step 1: Create the file**

```c
#include "vmm.h"
#include "pmm.h"     /* PAGE_SIZE, pmm_alloc_page() */
#include "arch.h"    /* arch_vmm_load_pml4, arch_vmm_invlpg, ARCH_KERNEL_VIRT_BASE */
#include "printk.h"
#include <stdint.h>

#define KERN_VMA      ARCH_KERNEL_VIRT_BASE

#define PML4_IDX(v)   (((v) >> 39) & 0x1FF)
#define PDPT_IDX(v)   (((v) >> 30) & 0x1FF)
#define PD_IDX(v)     (((v) >> 21) & 0x1FF)
#define PT_IDX(v)     (((v) >> 12) & 0x1FF)

#define PTE_PS        (1UL << 7)           /* huge page flag (PD level) */
#define PTE_ADDR_MASK 0x000FFFFFFFFFF000UL /* physical address bits in a PTE */

/* Physical address of the active PML4 — set by vmm_init(), used by
 * vmm_map_page() and vmm_unmap_page(). */
static uint64_t s_pml4_phys;

/* --------------------------------------------------------------------------
 * Internal helpers
 * -------------------------------------------------------------------------- */

/*
 * zero_page — zero 512 qwords at a physical address.
 *
 * Safe while the identity window [0..2MB) is active: physical == virtual.
 * CONSTRAINT: phys must be < 2MB. All PMM allocations in Phase 3 satisfy
 * this because the kernel + bitmap fit well within 2MB on a 128MB QEMU
 * machine. If the kernel image ever grows past 2MB, or if vmm_map_page()
 * is called with a deep mapping that triggers a PMM allocation above 2MB,
 * this will write to an unmapped address and fault. Phase 4, which tears
 * down or expands the identity map, must replace this with a mapped-window
 * allocator.
 */
static void zero_page(uint64_t phys)
{
    volatile uint64_t *p = (volatile uint64_t *)(uintptr_t)phys;
    for (int i = 0; i < 512; i++)
        p[i] = 0;
}

/* Allocate one page from PMM, zero it, and return its physical address.
 * Panics on OOM. */
static uint64_t alloc_table(void)
{
    uint64_t phys = pmm_alloc_page();
    if (phys == 0) {
        printk("[VMM] FAIL: OOM allocating page table\n");
        for (;;) {}
    }
    zero_page(phys);
    return phys;
}

/* Return a pointer to a page table given its physical address.
 * Valid only while the identity mapping is active (Phase 3). */
static uint64_t *phys_to_table(uint64_t phys)
{
    return (uint64_t *)(uintptr_t)phys;
}

/* Walk to a child table entry in parent[idx].
 * If not present, allocate and install a new table.
 * Returns the physical address of the child table. */
static uint64_t ensure_table(uint64_t *parent, uint64_t idx)
{
    if (!(parent[idx] & VMM_FLAG_PRESENT)) {
        uint64_t child_phys = alloc_table();
        parent[idx] = child_phys | VMM_FLAG_PRESENT | VMM_FLAG_WRITABLE;
        return child_phys;
    }
    return parent[idx] & PTE_ADDR_MASK;
}

/* --------------------------------------------------------------------------
 * Public API
 * -------------------------------------------------------------------------- */

void vmm_init(void)
{
    /* Step 1: allocate and zero the PML4. */
    s_pml4_phys = alloc_table();
    uint64_t *pml4 = phys_to_table(s_pml4_phys);

    /* Step 2: identity window [0..2MB) -> physical [0..2MB).
     * Keeps boot-time physical pointers valid after the cr3 switch. */
    uint64_t pdpt_lo_phys = alloc_table();
    uint64_t pd_lo_phys   = alloc_table();
    uint64_t *pdpt_lo = phys_to_table(pdpt_lo_phys);
    uint64_t *pd_lo   = phys_to_table(pd_lo_phys);

    pml4[0]    = pdpt_lo_phys | VMM_FLAG_PRESENT | VMM_FLAG_WRITABLE;
    pdpt_lo[0] = pd_lo_phys   | VMM_FLAG_PRESENT | VMM_FLAG_WRITABLE;
    pd_lo[0]   = 0x0UL        | VMM_FLAG_PRESENT | VMM_FLAG_WRITABLE | PTE_PS;

    /* Step 3: kernel window [KERN_VMA..KERN_VMA+2MB) -> physical [0..2MB).
     * PML4 index = 511 (bits 47:39 of KERN_VMA).
     * PDPT index = 510 (bits 38:30 of KERN_VMA). */
    uint64_t pdpt_hi_phys = alloc_table();
    uint64_t pd_hi_phys   = alloc_table();
    uint64_t *pdpt_hi = phys_to_table(pdpt_hi_phys);
    uint64_t *pd_hi   = phys_to_table(pd_hi_phys);

    pml4[PML4_IDX(KERN_VMA)]    = pdpt_hi_phys | VMM_FLAG_PRESENT | VMM_FLAG_WRITABLE;
    pdpt_hi[PDPT_IDX(KERN_VMA)] = pd_hi_phys   | VMM_FLAG_PRESENT | VMM_FLAG_WRITABLE;
    pd_hi[PD_IDX(KERN_VMA)]     = 0x0UL        | VMM_FLAG_PRESENT | VMM_FLAG_WRITABLE | PTE_PS;

    /* Step 4: switch cr3 to new page tables. */
    arch_vmm_load_pml4(s_pml4_phys);

    /* Step 5: report. */
    printk("[VMM] OK: kernel mapped to 0xFFFFFFFF80000000\n");
}

void vmm_map_page(uint64_t virt, uint64_t phys, uint64_t flags)
{
    if (virt & (PAGE_SIZE - 1)) {
        printk("[VMM] FAIL: vmm_map_page unaligned virt\n");
        for (;;) {}
    }
    if (phys & (PAGE_SIZE - 1)) {
        printk("[VMM] FAIL: vmm_map_page unaligned phys\n");
        for (;;) {}
    }

    uint64_t *pml4    = phys_to_table(s_pml4_phys);
    uint64_t pdpt_phys = ensure_table(pml4,              PML4_IDX(virt));
    uint64_t pd_phys   = ensure_table(phys_to_table(pdpt_phys), PDPT_IDX(virt));
    uint64_t pt_phys   = ensure_table(phys_to_table(pd_phys),   PD_IDX(virt));
    uint64_t *pt       = phys_to_table(pt_phys);
    uint64_t  idx      = PT_IDX(virt);

    if (pt[idx] & VMM_FLAG_PRESENT) {
        printk("[VMM] FAIL: vmm_map_page double-map\n");
        for (;;) {}
    }
    pt[idx] = (phys & PTE_ADDR_MASK) | flags | VMM_FLAG_PRESENT;
}

void vmm_unmap_page(uint64_t virt)
{
    if (virt & (PAGE_SIZE - 1)) {
        printk("[VMM] FAIL: vmm_unmap_page unaligned virt\n");
        for (;;) {}
    }

    uint64_t *pml4 = phys_to_table(s_pml4_phys);
    if (!(pml4[PML4_IDX(virt)] & VMM_FLAG_PRESENT)) {
        printk("[VMM] FAIL: vmm_unmap_page not mapped\n");
        for (;;) {}
    }
    uint64_t *pdpt = phys_to_table(pml4[PML4_IDX(virt)] & PTE_ADDR_MASK);
    if (!(pdpt[PDPT_IDX(virt)] & VMM_FLAG_PRESENT)) {
        printk("[VMM] FAIL: vmm_unmap_page not mapped\n");
        for (;;) {}
    }
    uint64_t *pd = phys_to_table(pdpt[PDPT_IDX(virt)] & PTE_ADDR_MASK);
    if (!(pd[PD_IDX(virt)] & VMM_FLAG_PRESENT)) {
        printk("[VMM] FAIL: vmm_unmap_page not mapped\n");
        for (;;) {}
    }
    uint64_t *pt = phys_to_table(pd[PD_IDX(virt)] & PTE_ADDR_MASK);
    if (!(pt[PT_IDX(virt)] & VMM_FLAG_PRESENT)) {
        printk("[VMM] FAIL: vmm_unmap_page not mapped\n");
        for (;;) {}
    }
    pt[PT_IDX(virt)] = 0;
    arch_vmm_invlpg(virt);
}
```

## Task 9: Rewrite boot.asm

**Files:**
- Rewrite: `kernel/arch/x86_64/boot.asm`

This is the most invasive task. Key changes:
1. GDT moves from `section .data` to `section .text.boot` (must have physical VMA=LMA)
2. `_start` (32-bit code) moves to `section .text.boot`
3. All `.bss` label accesses in 32-bit code use `(label - KERN_VMA)` for physical addresses
4. Page table setup extended from 3 tables to 5 (add `pdpt_hi`, `pd_hi`)
5. PDPT index for higher-half = `PDPT_HI_IDX equ ((KERN_VMA >> 30) & 0x1FF)` = 510 (derived, not magic)
6. Far jump uses global `long_mode_phys` (in `.text.boot`, VMA=LMA=physical) not a local label
7. After far jump: absolute `jmp rax` to `long_mode_high` (in `section .text`, higher-half VMA)
8. Segment register setup + stack + `kernel_main` call move to `long_mode_high`

- [ ] **Step 1: Replace the entire file**

```nasm
; boot.asm — Multiboot2 header, 32-to-64 long mode setup, kernel entry
;
; Entry state per multiboot2 spec:
;   CPU: 32-bit protected mode, interrupts disabled, paging off
;   EAX: multiboot2 magic value (0x36D76289)
;   EBX: physical address of multiboot2 info structure
;   Stack: undefined — we set it up before any C call
;
; This file is x86-specific and intentionally lives in kernel/arch/x86_64/.

MULTIBOOT2_MAGIC  equ 0xE85250D6   ; identifies this as a multiboot2 header
MULTIBOOT2_ARCH   equ 0             ; 0 = i386 (32-bit protected mode entry)
KERNEL_STACK_SIZE equ 0x4000        ; 16KB boot stack

KERN_VMA     equ 0xFFFFFFFF80000000  ; higher-half virtual base for kernel sections
; PDPT index for KERN_VMA: bits 38:30. Derived from KERN_VMA — do NOT hardcode 510.
; If KERN_VMA ever changes, this index follows automatically.
PDPT_HI_IDX  equ ((KERN_VMA >> 30) & 0x1FF)   ; = 510


; ─── Multiboot2 header ────────────────────────────────────────────────────────
; Must land within the first 8KB of the ELF image so GRUB finds it.
; Linker script forces .multiboot to be the very first section.
section .multiboot
align 8
multiboot_header_start:
    dd MULTIBOOT2_MAGIC
    dd MULTIBOOT2_ARCH
    dd (multiboot_header_end - multiboot_header_start)
    ; Checksum: magic + arch + length + checksum must sum to 0 (mod 2^32)
    dd -(MULTIBOOT2_MAGIC + MULTIBOOT2_ARCH + \
         (multiboot_header_end - multiboot_header_start))
    ; End tag (type=0, flags=0, size=8)
    dw 0
    dw 0
    dd 8
multiboot_header_end:


; ─── Boot trampoline section ──────────────────────────────────────────────────
; VMA = LMA = physical for everything in this section.
; This includes the GDT (lgdt runs in 32-bit mode and needs a physical address),
; _start (32-bit setup code), and the early 64-bit trampoline (long_mode_phys).
section .text.boot


; ─── GDT for 64-bit mode ──────────────────────────────────────────────────────
; Must be in .text.boot so its VMA = LMA = physical address.
; lgdt [gdt64_ptr] runs in 32-bit mode before paging — it must reference a
; physical address. Putting the GDT in .data would give it a higher-half VMA
; (after the linker VMA/LMA split), making the lgdt operand wrong.
align 8
gdt64:
    ; Entry 0: null descriptor (required)
    dq 0x0000000000000000
    ; Entry 1 (selector 0x08): 64-bit code segment
    ;   Access: P=1, DPL=0, S=1, Type=1010 (code/execute/read)
    ;   Flags: G=1, L=1 (64-bit), D=0
    dq 0x00AF9A000000FFFF
    ; Entry 2 (selector 0x10): 64-bit data segment
    ;   Access: P=1, DPL=0, S=1, Type=0010 (data/read/write)
    dq 0x00AF92000000FFFF
gdt64_end:

; LGDT descriptor: 2-byte limit + 4-byte base.
; gdt64 is in .text.boot so its address IS the physical address.
gdt64_ptr:
    dw (gdt64_end - gdt64 - 1)
    dd gdt64                        ; physical address of GDT (VMA=LMA in .text.boot)


; ─── Entry point ──────────────────────────────────────────────────────────────
bits 32

; _start — first instruction executed after GRUB hands off control.
;
; Purpose: transition from 32-bit protected mode to 64-bit long mode, build
;          page tables for identity [0..2MB) and kernel higher-half window,
;          enable paging, then jump to long_mode_phys (physical) and onward
;          to long_mode_high (higher-half virtual).
; Clobbers: all registers
; Calling convention: none (we are the entry point)
global _start
_start:
    cli                         ; disable interrupts (undefined state from bootloader)

    ; Preserve multiboot2 args before we clobber EAX/EBX.
    ; RDI = mb_magic (first System V AMD64 arg), RSI = mb_info (second arg).
    mov edi, eax                ; mb_magic → will be first  arg (RDI)
    mov esi, ebx                ; mb_info  → will be second arg (RSI); < 4GB physical

    ; ── Load 64-bit GDT (still in 32-bit mode) ──────────────────────────
    ; gdt64_ptr is in .text.boot — its address is physical. Correct here.
    lgdt [gdt64_ptr]

    ; ── Enable Physical Address Extension (required for long mode) ───────
    mov eax, cr4
    or  eax, (1 << 5)           ; CR4.PAE
    mov cr4, eax

    ; ── Build page tables ─────────────────────────────────────────────────────
    ;
    ; All page table labels are in .bss. After the linker VMA/LMA split, .bss
    ; labels have higher-half VMAs (0xFFFFFFFF80xxxxxx). In 32-bit mode we need
    ; their PHYSICAL addresses. NASM evaluates (label - KERN_VMA) as a constant
    ; expression at link time; the result is the physical address and fits in 32 bits.
    ;
    ; Layout:
    ;   pml4[0]              → pdpt_lo  (identity: first 512GB)
    ;   pml4[511]            → pdpt_hi  (higher-half: top 512GB)
    ;   pdpt_lo[0]           → pd_lo    (identity first 1GB)
    ;   pdpt_hi[PDPT_HI_IDX] → pd_hi    (covers 0xFFFFFFFF80000000)
    ;   pd_lo[0]             → 2MB huge page @ physical 0  (identity window)
    ;   pd_hi[0]             → 2MB huge page @ physical 0  (kernel window)
    ;
    ; All paging is off during this setup — writes are direct physical accesses.

    ; pml4[0] = physical(pdpt_lo) | present | writable
    mov eax, (pdpt_lo - KERN_VMA)
    or  eax, 0x3
    mov [pml4_table - KERN_VMA], eax

    ; pml4[511] = physical(pdpt_hi) | present | writable
    ; Index 511 * 8 = 4088 = 0xFF8 bytes into the table
    mov eax, (pdpt_hi - KERN_VMA)
    or  eax, 0x3
    mov [pml4_table - KERN_VMA + 0xFF8], eax

    ; pdpt_lo[0] = physical(pd_lo) | present | writable
    mov eax, (pd_lo - KERN_VMA)
    or  eax, 0x3
    mov [pdpt_lo - KERN_VMA], eax

    ; pdpt_hi[PDPT_HI_IDX] = physical(pd_hi) | present | writable
    ; Byte offset = PDPT_HI_IDX * 8 = 510 * 8 = 4080
    mov eax, (pd_hi - KERN_VMA)
    or  eax, 0x3
    mov [pdpt_hi - KERN_VMA + PDPT_HI_IDX * 8], eax

    ; pd_lo[0] = 2MB huge page at physical 0 (present | writable | PS)
    mov dword [pd_lo - KERN_VMA], 0x83

    ; pd_hi[0] = 2MB huge page at physical 0 (present | writable | PS)
    mov dword [pd_hi - KERN_VMA], 0x83

    ; Load PML4 physical address into CR3
    mov eax, (pml4_table - KERN_VMA)
    mov cr3, eax

    ; ── Set EFER.LME (Long Mode Enable) via MSR 0xC0000080 ───────────────
    mov ecx, 0xC0000080
    rdmsr
    or  eax, (1 << 8)           ; EFER.LME
    wrmsr

    ; ── Enable paging (activates compatibility mode) ─────────────────────
    ; Also set CR0.WP (write protect) so ring-0 respects read-only pages.
    mov eax, cr0
    or  eax, (1 << 31) | (1 << 16)   ; CR0.PG | CR0.WP
    mov cr0, eax

    ; CPU is now in IA-32e compatibility submode.
    ; Far jump to 64-bit code selector activates true 64-bit mode.
    ; long_mode_phys is in .text.boot (VMA = LMA = physical). The 32-bit
    ; far-jump offset is physical — valid because the identity window covers it.
    jmp 0x08:long_mode_phys


; ── 64-bit trampoline (still in .text.boot, identity-mapped) ─────────────────
bits 64

; long_mode_phys — executes at physical address, identity-mapped.
; Purpose: jump to the higher-half virtual address of long_mode_high.
; Global so it can be used as the far-jump target above.
global long_mode_phys
long_mode_phys:
    ; long_mode_high is in section .text (VMA = higher-half 0xFFFFFFFF80xxxxxx).
    ; mov rax encodes the full 64-bit virtual address; jmp rax crosses the gap.
    mov rax, long_mode_high
    jmp rax


; ─── Higher-half kernel entry ─────────────────────────────────────────────────
; VMA = higher-half. Executing here means the kernel window is live.
; All C symbols (.text/.rodata/.data/.bss) are at correct virtual addresses.
section .text
bits 64

global long_mode_high
long_mode_high:
    ; ── Load 64-bit data segment into DS, ES, SS ─────────────────────────
    mov ax, 0x10
    mov ds, ax
    mov es, ax
    mov ss, ax
    xor ax, ax
    mov fs, ax
    mov gs, ax

    ; ── Set up the boot stack ─────────────────────────────────────────────
    ; boot_stack_top is in .bss (higher-half VMA) — correct virtual address.
    ; 16-byte aligned as required by the System V AMD64 ABI.
    mov rsp, boot_stack_top

    ; ── Call kernel_main(uint32_t mb_magic, void *mb_info) ───────────────
    ; RDI = mb_magic, RSI = mb_info — set at _start, untouched since then.
    extern kernel_main
    call kernel_main

    ; ── Hang if kernel_main returns (it must not) ─────────────────────────
.hang:
    hlt
    jmp .hang


; ─── BSS: page tables and boot stack ─────────────────────────────────────────
; The bootloader zeros .bss before calling _start.
; After the VMA/LMA split, all labels here are higher-half virtual addresses.
; Use (label - KERN_VMA) to get physical addresses in 32-bit boot code.
section .bss
align 4096
pml4_table: resb 4096
align 4096
pdpt_lo:    resb 4096   ; identity PDPT (was pdpt_table)
align 4096
pdpt_hi:    resb 4096   ; higher-half PDPT (new)
align 4096
pd_lo:      resb 4096   ; identity PD (was pd_table)
align 4096
pd_hi:      resb 4096   ; higher-half PD (new)

align 16
boot_stack:     resb KERNEL_STACK_SIZE
boot_stack_top:
```

```c
#include "arch.h"
#include "printk.h"
#include "cap.h"
#include "pmm.h"
#include "vmm.h"

/*
 * kernel_main — top-level kernel entry point.
 *
 * Called from boot.asm after long mode is established, paging is active,
 * and execution has jumped to the higher-half virtual address.
 * Arguments follow the System V AMD64 ABI (set up in boot.asm).
 *
 * mb_magic: multiboot2 magic value (0x36D76289). Ignored in Phase 3;
 *           validated in a future phase when we need to abort on bad boot.
 *
 * mb_info:  physical address of the multiboot2 info structure.
 *           arch_mm_init() consumes this before vmm_init() runs.
 *           It is a physical address; the identity window keeps it accessible.
 */
void kernel_main(uint32_t mb_magic, void *mb_info)
{
    (void)mb_magic;

    arch_init();           /* serial_init + vga_init                    */
    arch_mm_init(mb_info); /* parse multiboot2 memory map               */
    pmm_init();            /* bitmap allocator — [PMM] OK               */
    vmm_init();            /* page tables, cr3 switch — [VMM] OK        */
    cap_init();            /* [CAP] OK                                  */
    printk("[AEGIS] System halted.\n");
    arch_debug_exit(0x01); /* QEMU exits with code 3                    */
    for (;;) {}            /* if not running in QEMU                    */
}
```
