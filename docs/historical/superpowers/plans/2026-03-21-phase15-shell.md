**Tech Stack:** x86-64 kernel C/NASM, musl-gcc static, xxd for binary embedding.

### Background

`sys_execve` needs to redirect the process to a new RIP and RSP. The `syscall_entry.asm` landing pad saves user state onto the kernel stack manually. We expose a pointer to that saved state as a `syscall_frame_t *` passed as the **first** SysV argument to `syscall_dispatch`. All existing arguments shift by one position.

The `syscall_frame_t` field order must exactly match the push order in `syscall_entry.asm`. The last-pushed value is at the lowest address (struct base). Current push sequence (before this change):
1. `push [g_user_rsp]` (user RSP — deepest, highest address in frame)
2. `push rcx` (user RIP)
3. `push r11` (RFLAGS)
4. `push r8`
5. `push r9`
6. `push r10` (shallowest — lowest address = struct base)

After discarding the arg6 slot (`add rsp, 8`), `frame = RSP + 8` points at the saved `r10`.

- [ ] **Step 2: Implement `sched_block` in `sched.c`**

```c
void
sched_block(void)
{
    aegis_task_t *old = s_current;
    aegis_task_t *prev;

    old->state = TASK_BLOCKED;

    /* Unlink old from the circular run queue.
     * Walk to find the node whose next == old. */
    prev = old;
    while (prev->next != old)
        prev = prev->next;
    prev->next = old->next;  /* splice out old */

    /* Advance s_current past the unlinked task; skip non-RUNNING tasks.
     * task_idle guarantees the loop terminates. */
    s_current = old->next;
    while (s_current->state != TASK_RUNNING)
        s_current = s_current->next;

    ctx_switch(old, s_current);
}
```

```c
void
sched_exit(void)
{
    /* ── Deferred cleanup from the PREVIOUS exiting kernel/orphaned task ──
     * This block must remain at the TOP of sched_exit, before any new exit
     * logic. It frees the TCB + kernel stack of the task that exited last
     * time (the task that was running when g_prev_dying_tcb was set).
     * This block is the existing deferred-cleanup pattern; preserve it verbatim. */
    if (g_prev_dying_tcb) {
        kva_free_pages(g_prev_dying_tcb->stack_base, g_prev_dying_tcb->stack_pages);
        kva_free_pages(g_prev_dying_tcb, 1);
        g_prev_dying_tcb = NULL;
    }

    /* Switch to master PML4 so kernel structures are safely accessible. */
    vmm_switch_to(vmm_get_master_pml4());

    if (s_current->is_user) {
        aegis_process_t *dying = (aegis_process_t *)s_current;
        /* dying->exit_status was set by sys_exit before calling sched_exit. */

        /* Mark self zombie — stays in run queue until waitpid reaps. */
        s_current->state = TASK_ZOMBIE;

        /* Wake blocked parent waiting for this child, if any. */
        aegis_task_t *t = s_current->next;
        while (t != s_current) {
            if (t->is_user && t->state == TASK_BLOCKED) {
                aegis_process_t *p = (aegis_process_t *)t;
                if (p->pid == dying->ppid &&
                    (t->waiting_for == 0 || t->waiting_for == dying->pid)) {
                    sched_wake(t);
                    break;
                }
            }
            t = t->next;
        }

        /* Shutdown detection: scan for remaining RUNNING user tasks.
         * Zombies have is_user=1 but state==TASK_ZOMBIE — they are not live.
         * Without the state check, this scan would never trigger shutdown
         * because the just-zombified task is still in the queue with is_user=1. */
        int live_users = 0;
        t = s_current->next;
        while (t != s_current) {
            if (t->is_user && t->state != TASK_ZOMBIE)
                live_users = 1;
            t = t->next;
        }
        if (!live_users) {
            arch_request_shutdown();
        }

        /* Yield — zombie stays in queue until waitpid reaps.
         * Do NOT call the deferred-cleanup path; that is for non-zombie exits. */
        sched_yield_to_next();
        /* unreachable — zombie never resumes */
        for (;;) {}
    }

    /* Kernel task exit path (existing deferred-cleanup logic) */
    /* ... (preserve existing non-user sched_exit code) ... */
}
```

```c
 * Returns the VA of the mapped page.

 * across any call that may call vmm_window_map/map2 internally. */

vmm_window_map2(uint64_t phys)
{

    __asm__ volatile("invlpg [%0]" : : "r"(VMM_WINDOW_VA + 4096) : "memory");
}
```

- [ ] **Step 2: Implement `vmm_copy_user_pages` in `vmm.c`**

```c
int
vmm_copy_user_pages(uint64_t src_pml4, uint64_t dst_pml4)
{
    /* Walk PML4 entries 0–255 of src_pml4 using the window allocator. */
    uint64_t pml4i, pdpti, pdi, pti;
    for (pml4i = 0; pml4i < 256; pml4i++) {
        uint64_t *pml4 = vmm_window_map(src_pml4);
        uint64_t pml4e = pml4[pml4i];
        vmm_window_unmap();
        if (!(pml4e & VMM_FLAG_PRESENT)) continue;

        uint64_t pdpt_phys = pml4e & ~0xFFFULL;
        for (pdpti = 0; pdpti < 512; pdpti++) {
            uint64_t *pdpt = vmm_window_map(pdpt_phys);
            uint64_t pdpte = pdpt[pdpti];
            vmm_window_unmap();
            if (!(pdpte & VMM_FLAG_PRESENT)) continue;

            uint64_t pd_phys = pdpte & ~0xFFFULL;
            for (pdi = 0; pdi < 512; pdi++) {
                uint64_t *pd = vmm_window_map(pd_phys);
                uint64_t pde = pd[pdi];
                vmm_window_unmap();
                if (!(pde & VMM_FLAG_PRESENT)) continue;

                uint64_t pt_phys = pde & ~0xFFFULL;
                for (pti = 0; pti < 512; pti++) {
                    uint64_t *pt = vmm_window_map(pt_phys);
                    uint64_t pte = pt[pti];
                    vmm_window_unmap();
                    if (!(pte & VMM_FLAG_PRESENT)) continue;
                    if (!(pte & VMM_FLAG_USER)) continue;

                    uint64_t src_phys = pte & ~0x8000000000000FFFULL;
                    /* Preserve all PTE flags including bit 63 (NX/XD bit). */
                    uint64_t flags    = pte &  0x8000000000000FFFULL;

                    /* Allocate destination frame */
                    uint64_t dst_phys = pmm_alloc_page();
                    if (!dst_phys) return -1;

                    /* Copy via two window slots */
                    void *src_va = vmm_window_map(src_phys);
                    void *dst_va = vmm_window_map2(dst_phys);
                    __builtin_memcpy(dst_va, src_va, 4096);
                    vmm_window_unmap2();
                    vmm_window_unmap();

                    /* Reconstruct virtual address */
                    uint64_t va = (pml4i << 39) | (pdpti << 30) |
                                  (pdi   << 21) | (pti   << 12);
                    /* Sign-extend if bit 47 is set (user space — it won't be,
                     * since pml4i < 256, but be explicit) */

                    vmm_map_user_page(dst_pml4, va, dst_phys, flags);
                }
            }
        }
    }
    return 0;
}
```

```c
        uint64_t pdpt_phys = pml4e & ~0xFFFULL;
        for (pdpti = 0; pdpti < 512; pdpti++) {
            uint64_t *pdpt = vmm_window_map(pdpt_phys);

            vmm_window_unmap();

            uint64_t pd_phys = pdpte & ~0xFFFULL;
            for (pdi = 0; pdi < 512; pdi++) {

                uint64_t pde = pd[pdi];
                vmm_window_unmap();

                    uint64_t pte = pt[pti];
```

```c
    aegis_process_t *parent      = (aegis_process_t *)parent_task;

    /* 1. Allocate child PCB */
    aegis_process_t *child = kva_alloc_pages(1);
    if (!child) return (uint64_t)-ENOMEM;

    /* 2. Copy parent fields */
    uint32_t ci;
    for (ci = 0; ci < PROC_MAX_FDS; ci++)
        child->fds[ci] = parent->fds[ci];
    for (ci = 0; ci < CAP_TABLE_SIZE; ci++)
        child->caps[ci] = parent->caps[ci];
    child->brk       = parent->brk;
    child->mmap_base = parent->mmap_base;
    child->fs_base   = parent->fs_base;
    __builtin_memcpy(child->cwd, parent->cwd, sizeof(parent->cwd));
    child->pid             = proc_alloc_pid();
    child->ppid            = parent->pid;
    child->exit_status     = 0;
    child->task.state      = TASK_RUNNING;
    child->task.waiting_for = 0;
    child->task.is_user    = 1;
    child->task.tid        = child->pid;   /* use pid as tid */
    child->task.stack_pages = 1;

    /* 3. Create child PML4 */
    child->pml4_phys = vmm_create_user_pml4();

    /* 4. Copy parent user pages into child PML4 */
    if (vmm_copy_user_pages(parent->pml4_phys, child->pml4_phys) != 0) {
        kva_free_pages(child, 1);
        return (uint64_t)-ENOMEM;
    }

    /* 5. Allocate child kernel stack */
    uint8_t *kstack = kva_alloc_pages(1);
    if (!kstack) {
        vmm_free_user_pml4(child->pml4_phys);
        kva_free_pages(child, 1);
        return (uint64_t)-ENOMEM;
    }

    /* 6. Build child initial kernel stack frame.
     *
     * The frame (from low to high address) must be:
     *   [r15=0][r14=0][r13=0][r12=0][rbp=0][rbx=0]  ← ctx_switch callee-saves
     *   [fork_child_return]                           ← ret address for ctx_switch
     *   [r10 = frame->r10]                            ← syscall_frame_t +0
     *   [r9  = frame->r9]                             ← syscall_frame_t +8
     *   [r8  = frame->r8]                             ← syscall_frame_t +16
     *   [rflags = frame->rflags]                      ← syscall_frame_t +24
     *   [rip    = frame->rip]                         ← syscall_frame_t +32
     *   [user_rsp = frame->user_rsp]                  ← syscall_frame_t +40
     *
     * Push order: highest address first (push = decrement sp, then write).
     */
    uint64_t *sp = (uint64_t *)(kstack + 4096);
    *--sp = frame->user_rsp;
    *--sp = frame->rip;
    *--sp = frame->rflags;
    *--sp = frame->r8;
    *--sp = frame->r9;
    *--sp = frame->r10;
    *--sp = (uint64_t)(uintptr_t)fork_child_return;
    *--sp = 0;  /* rbx */
    *--sp = 0;  /* rbp */
    *--sp = 0;  /* r12 */
    *--sp = 0;  /* r13 */
    *--sp = 0;  /* r14 */
    *--sp = 0;  /* r15 ← task.rsp */

    child->task.rsp              = (uint64_t)(uintptr_t)sp;
    child->task.stack_base       = kstack;
    child->task.kernel_stack_top = (uint64_t)(uintptr_t)(kstack + 4096);

    /* Update TSS RSP0 for parent (it remains current) */
    arch_set_kernel_stack(parent_task->kernel_stack_top);

    /* 7. Add child to run queue */
    sched_add(&child->task);

    /* Return child PID to parent */
    return (uint64_t)child->pid;
}
```

## Task 9: `sys_execve` (syscall 59)

**Files:**
- Modify: `kernel/syscall/syscall.c`

- [ ] **Step 1: Implement `sys_execve`**

```c
static uint64_t
sys_execve(syscall_frame_t *frame,
           uint64_t path_uptr, uint64_t argv_uptr, uint64_t envp_uptr)
{
    (void)envp_uptr;  /* Phase 15: empty environment */

    aegis_process_t *proc = (aegis_process_t *)sched_current();

    /* 1. Copy path from user (≤255 bytes) */
    if (!user_ptr_valid(path_uptr, 1)) return (uint64_t)-EFAULT;
    char path[256];
    {
        /* Walk user memory byte-by-byte to find the null terminator safely. */
        uint64_t i;
        for (i = 0; i < sizeof(path) - 1; i++) {
            if (!user_ptr_valid(path_uptr + i, 1)) return (uint64_t)-EFAULT;
            char c;
            copy_from_user(&c, (void *)(uintptr_t)(path_uptr + i), 1);
            path[i] = c;
            if (c == '\0') break;
        }
        path[sizeof(path) - 1] = '\0';
    }

    /* 2. Copy argv from user (≤64 entries, each ≤255 bytes) */
    char  argv_bufs[64][256];
    char *argv[65];
    int   argc = 0;
    {
        uint64_t ptr_addr = argv_uptr;
        while (argc < 64) {
            if (!user_ptr_valid(ptr_addr, 8)) return (uint64_t)-EFAULT;
            uint64_t str_ptr;
            copy_from_user(&str_ptr, (void *)(uintptr_t)ptr_addr, 8);
            if (!str_ptr) break;  /* NULL terminator */
            /* Copy string */
            uint64_t i;
            for (i = 0; i < 255; i++) {
                if (!user_ptr_valid(str_ptr + i, 1)) return (uint64_t)-EFAULT;
                char c;
                copy_from_user(&c, (void *)(uintptr_t)(str_ptr + i), 1);
                argv_bufs[argc][i] = c;
                if (c == '\0') break;
            }
            argv_bufs[argc][255] = '\0';
            argv[argc] = argv_bufs[argc];
            argc++;
            ptr_addr += 8;
        }
        argv[argc] = (char *)0;
    }

    /* 3. Look up path in initrd */
    vfs_file_t f;
    if (initrd_open(path, &f) != 0) return (uint64_t)-ENOENT;

    /* 4. Free all user leaf pages; reuse PML4 */
    vmm_free_user_pages(proc->pml4_phys);
    vmm_switch_to(proc->pml4_phys);   /* reload CR3 to flush stale TLBs */

    /* 5. Reset heap/mmap state */
    proc->brk       = 0;
    proc->mmap_base = 0x0000700000000000ULL;
    proc->fs_base   = 0;

    /* 6. Load new ELF.
     * f.size is populated by initrd_open (via the new vfs_file_t.size field
     * added in Task 12). f.priv points to the initrd_entry_t; its data pointer
     * is accessed via a cast known to be safe here because initrd_open is the
     * only caller that sets priv for regular files, and its layout is fixed.
     * The cast is to (const uint8_t *)(uintptr_t)((const char **)f.priv)[1]
     * — but that is fragile. Instead we rely on f.size (populated by initrd_open)
     * and expose the data pointer by adding a vfs_ops_t.read_raw or simply
     * reading f.size bytes via the read op. For Phase 15, use a helper:
     *
     *   initrd_get_data(f.priv) — declared in initrd.h, returns const uint8_t* */
    extern const uint8_t *initrd_get_data(void *priv);
    elf_load_result_t er;
    if (elf_load(proc->pml4_phys, initrd_get_data(f.priv), f.size, &er) != 0)
        return (uint64_t)-ENOEXEC;
    proc->brk = er.brk;

    /* 7. Allocate + map 4 user stack pages (16 KB).
     * Worst-case argv: 64 args × 256 bytes = 16 KB of strings alone.
     * A single 4 KB page is insufficient. 4 pages gives 16 KB of stack
     * space, enough for the combined strings + pointer table + auxv.
     * Bounds checking in step 8 guards against overflow at runtime. */
#define USER_STACK_TOP_EXEC   0x7FFFFFFF000ULL
#define USER_STACK_NPAGES     4ULL
#define USER_STACK_PAGE_EXEC  (USER_STACK_TOP_EXEC - USER_STACK_NPAGES * 4096ULL)
    {
        uint64_t pn;
        for (pn = 0; pn < USER_STACK_NPAGES; pn++) {
            uint64_t phys = pmm_alloc_page();
            if (!phys) return (uint64_t)-ENOMEM;
            vmm_zero_page(phys);
            vmm_map_user_page(proc->pml4_phys,
                              USER_STACK_PAGE_EXEC + pn * 4096ULL, phys,
                              VMM_FLAG_PRESENT | VMM_FLAG_USER | VMM_FLAG_WRITABLE);
        }
    }

    /* 8. Build ABI initial stack at USER_STACK_TOP_EXEC.
     *
     * Layout (RSP must be 16-byte aligned AFTER the implicit push of the
     * return address in _start; i.e. RSP % 16 == 8 when control enters _start):
     *
     *   [rsp]                = argc
     *   [rsp+8*(1..argc)]    = argv[0..argc-1] pointers
     *   [rsp+8*(argc+1)]     = NULL (argv terminator)
     *   [rsp+8*(argc+2)]     = NULL (envp[0] — empty env)
     *   [rsp+8*(argc+3..)]   = auxv pairs: AT_PHDR, phdr_va, AT_PHNUM, phdr_count,
     *                                       AT_PAGESZ, 4096, AT_ENTRY, entry, AT_NULL, 0
     *   ... string data above (packed below USER_STACK_TOP_EXEC) ...
     */
    uint64_t sp_va = USER_STACK_TOP_EXEC;

    /* Pack argv strings at the top of the 4-page stack region, working down.
     * Uses vmm_phys_of_user to resolve the physical page for each address.
     * A string may straddle a page boundary; the split-write path handles that. */
    uint64_t str_ptrs[64];
    {
        int i;
        for (i = argc - 1; i >= 0; i--) {
            uint64_t slen = 0;
            while (argv[i][slen]) slen++;
            slen++;  /* include null terminator */
            /* Bounds check: abort if strings overflow the stack region */
            if (sp_va - slen < USER_STACK_PAGE_EXEC) return (uint64_t)-E2BIG;
            sp_va -= slen;

            /* Copy in page-sized chunks to handle strings that cross page boundaries. */
            uint64_t copied = 0;
            while (copied < slen) {
                uint64_t cur_va   = sp_va + copied;
                uint64_t pg_va    = cur_va & ~0xFFFULL;
                uint64_t off_in_pg = cur_va & 0xFFFULL;
                uint64_t chunk    = 4096 - off_in_pg;  /* bytes remaining in this page */
                if (chunk > slen - copied) chunk = slen - copied;
                uint64_t phys_pg  = vmm_phys_of_user(proc->pml4_phys, pg_va);
                void *dst = (uint8_t *)vmm_window_map(phys_pg) + off_in_pg;
                __builtin_memcpy(dst, argv[i] + copied, chunk);
                vmm_window_unmap();
                copied += chunk;
            }
            str_ptrs[i] = sp_va;
        }
    }

    /* Align sp_va for the pointer table (8-byte aligned) */
    sp_va &= ~7ULL;

    /* Count pointer table entries:
     * argc + 1 (NULL) + 1 (envp NULL) + 10 (5 auxv pairs) + 1 (argc) = argc + 13 */
    uint64_t table_size = (uint64_t)(argc + 13) * 8;

    /* Ensure RSP % 16 == 8 on entry to _start */
    sp_va -= table_size;
    if ((sp_va % 16) != 8)
        sp_va -= 8;
    if (sp_va < USER_STACK_PAGE_EXEC) return (uint64_t)-E2BIG;

    /* Helper: write one uint64_t into the user stack via window allocator.
     * Uses vmm_phys_of_user to locate the physical page for any address
     * in the 4-page stack region — handles cross-page table writes correctly. */
#define WRITE_STACK(va, val) do { \
    uint64_t _phys = vmm_phys_of_user(proc->pml4_phys, (va) & ~0xFFFULL); \
    uint64_t _off  = (va) & 0xFFFULL; \
    uint64_t *_p   = (uint64_t *)((uint8_t *)vmm_window_map(_phys) + _off); \
    *_p = (uint64_t)(val); \
    vmm_window_unmap(); \
} while(0)

    uint64_t wp = sp_va;
    WRITE_STACK(wp, (uint64_t)argc); wp += 8;
    {
        int i;
        for (i = 0; i < argc; i++) { WRITE_STACK(wp, str_ptrs[i]); wp += 8; }
    }
    WRITE_STACK(wp, 0ULL); wp += 8;  /* argv NULL */
    WRITE_STACK(wp, 0ULL); wp += 8;  /* envp NULL */
    /* auxv */
    WRITE_STACK(wp, 3ULL);          wp += 8;  /* AT_PHDR */
    WRITE_STACK(wp, er.phdr_va);    wp += 8;
    WRITE_STACK(wp, 5ULL);          wp += 8;  /* AT_PHNUM */
    WRITE_STACK(wp, (uint64_t)er.phdr_count); wp += 8;
    WRITE_STACK(wp, 6ULL);          wp += 8;  /* AT_PAGESZ */
    WRITE_STACK(wp, 4096ULL);       wp += 8;
    WRITE_STACK(wp, 9ULL);          wp += 8;  /* AT_ENTRY */
    WRITE_STACK(wp, er.entry);      wp += 8;
    WRITE_STACK(wp, 0ULL);          wp += 8;  /* AT_NULL */
    WRITE_STACK(wp, 0ULL);                    /* AT_NULL value */

    /* 9. Redirect sysret to new ELF */
    frame->rip      = er.entry;
    frame->user_rsp = sp_va;

    return 0;  /* not reached in practice (sysret goes to new ELF) */
}
```

Note: the WRITE_STACK macro maps and unmaps the stack page for each 8-byte write. This is correct (and safe) but inefficient. An optimisation would be to hold the window open for the entire table write and do direct writes — but the macro approach is clearer and correctness matters more here.

```c
#define WNOHANG  1

static uint64_t
sys_waitpid(uint64_t pid_arg, uint64_t wstatus_ptr, uint64_t options)
{
    aegis_process_t *caller = (aegis_process_t *)sched_current();
    int32_t          pid    = (int32_t)(uint32_t)pid_arg;

retry:;
    /* Scan run queue for a zombie child matching the request. */
    aegis_task_t *t = sched_current()->next;
    while (t != sched_current()) {
        if (t->is_user && t->state == TASK_ZOMBIE) {
            aegis_process_t *child = (aegis_process_t *)t;
            if (child->ppid == caller->pid &&
                (pid == -1 || (uint32_t)pid == child->pid)) {
                /* Found a zombie to reap. */
                uint32_t child_pid = child->pid;
                uint64_t status    = child->exit_status & 0xFF;

                /* Write exit status to user if requested */
                if (wstatus_ptr) {
                    if (!user_ptr_valid(wstatus_ptr, 4)) return (uint64_t)-EFAULT;
                    uint32_t wstatus_val = (uint32_t)(status << 8);
                    copy_to_user((void *)(uintptr_t)wstatus_ptr,
                                 &wstatus_val, 4);
                }

                /* Remove zombie from run queue */
                aegis_task_t *prev = t;
                while (prev->next != t) prev = prev->next;
                prev->next = t->next;

                /* Free zombie resources */
                kva_free_pages(child->task.stack_base, child->task.stack_pages);
                vmm_free_user_pml4(child->pml4_phys);
                kva_free_pages(child, 1);

                /* Clear waiting_for on the caller — it is no longer blocked */
                sched_current()->waiting_for = 0;

                return (uint64_t)child_pid;
            }
        }
        t = t->next;
    }

    /* No zombie found */
    if (options & WNOHANG) return 0;

    /* Block until a child changes state */
    sched_current()->waiting_for = (pid == -1) ? 0 : (uint32_t)pid;
    sched_block();
    goto retry;
}
```

- [ ] **Step 4: Implement `sys_getdents64` in `syscall.c`**

```c
/* linux_dirent64 matches the Linux kernel structure exactly. */
typedef struct {
    uint64_t d_ino;
    int64_t  d_off;
    uint16_t d_reclen;
    uint8_t  d_type;
    char     d_name[1];  /* flexible — actual size determined by d_reclen */
} __attribute__((packed)) linux_dirent64_t;

static uint64_t
sys_getdents64(uint64_t fd_num, uint64_t dirp, uint64_t count)
{
    aegis_process_t *proc = (aegis_process_t *)sched_current();
    if (fd_num >= PROC_MAX_FDS) return (uint64_t)-EBADF;
    vfs_file_t *f = &proc->fds[fd_num];
    if (!f->ops) return (uint64_t)-EBADF;
    if (!f->ops->readdir) return (uint64_t)-ENOTDIR;
    if (!user_ptr_valid(dirp, count)) return (uint64_t)-EFAULT;

    uint64_t written = 0;
    char name[256];
    uint8_t type;

    while (1) {
        if (f->ops->readdir(f->priv, f->offset, name, &type) != 0) break;

        /* Calculate record size: header(19) + strlen(name) + 1, round up to 8 */
        uint64_t namelen = 0;
        while (name[namelen]) namelen++;
        uint16_t reclen = (uint16_t)(19 + namelen + 1);
        reclen = (reclen + 7) & ~7;

        if (written + reclen > count) break;

        /* Build dirent in kernel buffer, then copy_to_user */
        uint8_t kbuf[300];
        linux_dirent64_t *d = (linux_dirent64_t *)kbuf;
        d->d_ino    = f->offset + 1;
        d->d_off    = (int64_t)(f->offset + 1);
        d->d_reclen = reclen;
        d->d_type   = type;
        uint64_t i;
        for (i = 0; i <= namelen; i++) d->d_name[i] = name[i];
        /* zero-pad to reclen */
        /* d_name occupies d_name[0..namelen] (namelen+1 bytes).
         * Pad d_name[namelen+1..reclen-20] to zero (reclen-19 total name bytes). */
        for (i = 1 + namelen; i < reclen - 19; i++) d->d_name[i] = '\0';

        copy_to_user((void *)(uintptr_t)(dirp + written), kbuf, reclen);
        written += reclen;
        f->offset++;
    }
    return written;
}
```

```c
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>
#include <fcntl.h>

#define MAX_ARGS 32
#define LINE_SIZE 256

static int
tokenize(char *line, char **argv, int max)
{
    int argc = 0;
    char *p = line;
    while (*p) {
        while (*p == ' ' || *p == '\t') p++;
        if (!*p || *p == '\n') break;
        if (argc >= max - 1) break;
        argv[argc++] = p;
        while (*p && *p != ' ' && *p != '\t' && *p != '\n') p++;
        if (*p) *p++ = '\0';
    }
    argv[argc] = NULL;
    return argc;
}

static void
do_cd(char **argv)
{
    const char *path = argv[1] ? argv[1] : "/";
    if (chdir(path) != 0)
        perror(path);
}

static void
do_help(void)
{
    puts("Builtins: cd [path], exit [n], help");
    puts("Other commands are looked up in /bin/");
}

int main(void)
{
    char  line[LINE_SIZE];
    char *argv[MAX_ARGS];
    char *envp[] = { NULL };

    printf("[SHELL] Aegis shell ready\n");
    for (;;) {
        write(1, "# ", 2);
        if (!fgets(line, sizeof(line), stdin))
            break;

        int argc = tokenize(line, argv, MAX_ARGS);
        if (argc == 0) continue;

        if (strcmp(argv[0], "exit") == 0) {
            int code = argv[1] ? atoi(argv[1]) : 0;
            exit(code);
        }
        if (strcmp(argv[0], "cd") == 0) {
            do_cd(argv);
            continue;
        }
        if (strcmp(argv[0], "help") == 0) {
            do_help();
            continue;
        }

        /* External command: fork + exec */
        pid_t pid = fork();
        if (pid == 0) {
            /* Child */
            char full[64];
            if (argv[0][0] != '/') {
                snprintf(full, sizeof(full), "/bin/%s", argv[0]);
                execve(full, argv, envp);
            } else {
                execve(argv[0], argv, envp);
            }
            fprintf(stderr, "%s: not found\n", argv[0]);
            _exit(127);
        } else if (pid > 0) {
            int status;
            waitpid(pid, &status, 0);
        } else {
            perror("fork");
        }
    }
    return 0;
}
```
