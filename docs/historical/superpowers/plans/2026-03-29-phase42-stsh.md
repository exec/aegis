**Goal:** Build stsh, the capability-aware management shell for Aegis — interactive line editing, history, tab completion, caps/sandbox builtins, paste detection, plus kernel support (new cap kinds, sys_cap_query syscall, sys_spawn cap_mask).

**Architecture:** Fork existing `/bin/sh` into `user/stsh/` with 9 source files. Add CAP_KIND_CAP_DELEGATE (13) and CAP_KIND_CAP_QUERY (14) to kernel cap system. Add sys_cap_query syscall (362). Extend sys_spawn with 5th cap_mask parameter. Update login chain to grant delegate/query caps to stsh.

**Tech Stack:** C (user-space, kernel), Rust (kernel/cap/src/lib.rs), musl libc (dynamic linking), QEMU q35 for testing.

Add at the end of the file (after sys_spawn):

```c
/*
 * sys_cap_query — syscall 362
 * Returns the capability table of a process.
 *   pid == 0: own caps (always allowed)
 *   pid != 0: target's caps (requires CAP_KIND_CAP_QUERY)
 * Copies cap_slot_t entries to user buffer.
 * Returns number of slots copied, or negative errno.
 */
uint64_t
sys_cap_query(uint64_t pid_arg, uint64_t buf_uptr, uint64_t buflen)
{
    aegis_process_t *caller = (aegis_process_t *)sched_current();
    if (!sched_current()->is_user)
        return (uint64_t)-(int64_t)1;  /* EPERM */

    aegis_process_t *target;

    if (pid_arg == 0) {
        /* Query own caps — always allowed */
        target = caller;
    } else {
        /* Query another process — requires CAP_KIND_CAP_QUERY */
        if (cap_check(caller->caps, CAP_TABLE_SIZE,
                      CAP_KIND_CAP_QUERY, CAP_RIGHTS_READ) != 0)
            return (uint64_t)-(int64_t)ENOCAP;

        /* Find target process by PID */
        target = NULL;
        aegis_task_t *t = sched_current()->next;
        while (t != sched_current()) {
            if (t->is_user) {
                aegis_process_t *p = (aegis_process_t *)t;
                if (p->pid == (uint32_t)pid_arg) {
                    target = p;
                    break;
                }
            }
            t = t->next;
        }
        if (!target)
            return (uint64_t)-(int64_t)3;  /* ESRCH */
    }

    /* Copy caps to user buffer */
    uint64_t copy_bytes = CAP_TABLE_SIZE * sizeof(cap_slot_t);
    if (buflen < copy_bytes)
        copy_bytes = buflen;

    uint64_t nslots = copy_bytes / sizeof(cap_slot_t);
    if (nslots == 0)
        return 0;

    copy_bytes = nslots * sizeof(cap_slot_t);

    if (copy_to_user((void *)buf_uptr, target->caps, copy_bytes) != 0)
        return (uint64_t)-(int64_t)14;  /* EFAULT */

    return nslots;
}
```

```c
    /* 12. Capabilities: baseline + exec_caps, OR cap_mask if provided. */
    {
        uint32_t ci;
        for (ci = 0; ci < CAP_TABLE_SIZE; ci++) {
            child->caps[ci].kind   = CAP_KIND_NULL;
            child->caps[ci].rights = 0;
        }

        if (cap_mask_uptr != 0) {
            /* cap_mask mode: caller must hold CAP_DELEGATE */
            if (cap_check(parent->caps, CAP_TABLE_SIZE,
                          CAP_KIND_CAP_DELEGATE, CAP_RIGHTS_READ) != 0) {
                /* Clean up and fail */
                kva_free_pages(kstack, 4);
                goto fail_child;
            }

            /* Copy cap_mask from userspace */
            cap_slot_t mask[CAP_TABLE_SIZE];
            if (copy_from_user(mask, (const void *)cap_mask_uptr,
                               sizeof(mask)) != 0) {
                kva_free_pages(kstack, 4);
                goto fail_child;
            }

            /* Validate and apply: caller can only grant caps it holds */
            for (ci = 0; ci < CAP_TABLE_SIZE; ci++) {
                if (mask[ci].kind == CAP_KIND_NULL)
                    continue;
                if (mask[ci].kind >= 16u) {
                    kva_free_pages(kstack, 4);
                    goto fail_child;
                }
                /* Caller must hold this cap to grant it */
                if (cap_check(parent->caps, CAP_TABLE_SIZE,
                              mask[ci].kind, mask[ci].rights) != 0) {
                    kva_free_pages(kstack, 4);
                    goto fail_child;
                }
                cap_grant(child->caps, CAP_TABLE_SIZE,
                          mask[ci].kind, mask[ci].rights);
            }

            /* exec_caps NOT applied in cap_mask mode */
        } else {
            /* Normal mode: baseline + exec_caps */
            cap_grant(child->caps, CAP_TABLE_SIZE, CAP_KIND_VFS_OPEN,   CAP_RIGHTS_READ);
            cap_grant(child->caps, CAP_TABLE_SIZE, CAP_KIND_VFS_WRITE,  CAP_RIGHTS_WRITE);
            cap_grant(child->caps, CAP_TABLE_SIZE, CAP_KIND_VFS_READ,   CAP_RIGHTS_READ);
            cap_grant(child->caps, CAP_TABLE_SIZE, CAP_KIND_NET_SOCKET, CAP_RIGHTS_READ);
            cap_grant(child->caps, CAP_TABLE_SIZE, CAP_KIND_THREAD_CREATE, CAP_RIGHTS_READ);
            cap_grant(child->caps, CAP_TABLE_SIZE, CAP_KIND_PROC_READ,  CAP_RIGHTS_READ | CAP_RIGHTS_WRITE);
            cap_grant(child->caps, CAP_TABLE_SIZE, CAP_KIND_FB,         CAP_RIGHTS_READ);

            /* Apply parent's pre-registered exec_caps, then consume them. */
            for (ci = 0; ci < CAP_TABLE_SIZE; ci++) {
                if (parent->exec_caps[ci].kind != CAP_KIND_NULL) {
                    cap_grant(child->caps, CAP_TABLE_SIZE,
                              parent->exec_caps[ci].kind,
                              parent->exec_caps[ci].rights);
                    parent->exec_caps[ci].kind   = CAP_KIND_NULL;
                    parent->exec_caps[ci].rights = 0;
                }
            }
        }

        /* Zero child's exec_caps. */
        for (ci = 0; ci < CAP_TABLE_SIZE; ci++) {
            child->exec_caps[ci].kind   = CAP_KIND_NULL;
            child->exec_caps[ci].rights = 0;
        }
    }
```

```c
#include "stsh.h"
#include <string.h>

int
parse_command(char *seg, cmd_t *cmd)
{
    cmd->argv[0]          = NULL;
    cmd->stdin_file       = NULL;
    cmd->stdout_file      = NULL;
    cmd->stderr_to_stdout = 0;

    int argc = 0;
    char *p  = seg;

    while (*p) {
        while (*p == ' ' || *p == '\t') p++;
        if (!*p || *p == '\n') break;

        if (p[0]=='2' && p[1]=='>' && p[2]=='&' && p[3]=='1') {
            cmd->stderr_to_stdout = 1;
            p += 4;
            continue;
        }
        if (p[0] == '>' && p[1] == '>') {
            p += 2;
            while (*p == ' ' || *p == '\t') p++;
            cmd->stdout_file = p;
            while (*p && *p != ' ' && *p != '\t' && *p != '\n') p++;
            if (*p) *p++ = '\0';
            continue;
        }
        if (*p == '>') {
            p++;
            while (*p == ' ' || *p == '\t') p++;
            cmd->stdout_file = p;
            while (*p && *p != ' ' && *p != '\t' && *p != '\n') p++;
            if (*p) *p++ = '\0';
            continue;
        }
        if (*p == '<') {
            p++;
            while (*p == ' ' || *p == '\t') p++;
            cmd->stdin_file = p;
            while (*p && *p != ' ' && *p != '\t' && *p != '\n') p++;
            if (*p) *p++ = '\0';
            continue;
        }

        if (argc >= MAX_ARGV) break;
        cmd->argv[argc++] = p;
        while (*p && *p != ' ' && *p != '\t' && *p != '\n') p++;
        if (*p) *p++ = '\0';
    }
    cmd->argv[argc] = NULL;
    return argc;
}

int
parse_pipeline(char *line, cmd_t *cmds, int max)
{
    int n    = 0;
    char *p  = line;
    char *seg = line;

    while (*p) {
        if (*p == '|') {
            *p++ = '\0';
            if (n < max) {
                if (parse_command(seg, &cmds[n]) > 0)
                    n++;
            }
            seg = p;
        } else {
            p++;
        }
    }
    if (n < max && parse_command(seg, &cmds[n]) > 0)
        n++;

    return n;
}
```

```c
#include "stsh.h"
#include <stdio.h>
#include <string.h>
#include <stdlib.h>

static char  s_env_store[ENV_MAX][256];  /* KEY=VALUE storage */
static char *s_env_ptrs[ENV_MAX + 1];   /* NULL-terminated for execve */
static int   s_env_count = 0;

void
env_init(char **envp)
{
    s_env_count = 0;
    if (envp) {
        for (int i = 0; envp[i] && s_env_count < ENV_MAX; i++) {
            strncpy(s_env_store[s_env_count], envp[i], 255);
            s_env_store[s_env_count][255] = '\0';
            s_env_ptrs[s_env_count] = s_env_store[s_env_count];
            s_env_count++;
        }
    }
    s_env_ptrs[s_env_count] = NULL;
}

const char *
env_get(const char *key)
{
    int klen = (int)strlen(key);
    for (int i = 0; i < s_env_count; i++) {
        if (strncmp(s_env_store[i], key, (size_t)klen) == 0 &&
            s_env_store[i][klen] == '=')
            return s_env_store[i] + klen + 1;
    }
    return NULL;
}

void
env_set(const char *key, const char *value)
{
    int klen = (int)strlen(key);

    /* Update existing */
    for (int i = 0; i < s_env_count; i++) {
        if (strncmp(s_env_store[i], key, (size_t)klen) == 0 &&
            s_env_store[i][klen] == '=') {
            snprintf(s_env_store[i], 256, "%s=%s", key, value);
            return;
        }
    }

    /* Add new */
    if (s_env_count < ENV_MAX) {
        snprintf(s_env_store[s_env_count], 256, "%s=%s", key, value);
        s_env_ptrs[s_env_count] = s_env_store[s_env_count];
        s_env_count++;
        s_env_ptrs[s_env_count] = NULL;
    }
}

void
env_print_all(void)
{
    for (int i = 0; i < s_env_count; i++)
        printf("%s\n", s_env_store[i]);
}

char **
env_as_array(void)
{
    return s_env_ptrs;
}

void
env_expand(const char *src, char *dst, int dstlen, int last_exit)
{
    int di = 0;
    const char *p = src;

    while (*p && di < dstlen - 1) {
        if (*p != '$') {
            dst[di++] = *p++;
            continue;
        }
        p++;  /* skip $ */

        if (*p == '?') {
            /* $? — last exit code */
            p++;
            char num[12];
            int nlen = snprintf(num, sizeof(num), "%d", last_exit);
            for (int i = 0; i < nlen && di < dstlen - 1; i++)
                dst[di++] = num[i];
            continue;
        }

        /* ${VAR} or $VAR */
        int braced = 0;
        if (*p == '{') { braced = 1; p++; }

        const char *start = p;
        while ((*p >= 'A' && *p <= 'Z') || (*p >= 'a' && *p <= 'z') ||
               (*p >= '0' && *p <= '9') || *p == '_')
            p++;

        int vlen = (int)(p - start);
        if (braced && *p == '}') p++;

        if (vlen > 0) {
            char key[64];
            if (vlen > 63) vlen = 63;
            memcpy(key, start, (size_t)vlen);
            key[vlen] = '\0';
            const char *val = env_get(key);
            if (val) {
                int slen = (int)strlen(val);
                for (int i = 0; i < slen && di < dstlen - 1; i++)
                    dst[di++] = val[i];
            }
        }
    }
    dst[di] = '\0';
}
```

```c
#include "stsh.h"
#include <stdio.h>
#include <string.h>

static char s_entries[HIST_SIZE][LINE_SIZE];
static int  s_count  = 0;   /* total entries written */
static int  s_cursor = -1;  /* navigation position; -1 = at bottom */
static int  s_privileged = 0;

static int
wrap(int idx)
{
    return idx % HIST_SIZE;
}

void
hist_init(int privileged)
{
    s_privileged = privileged;
    s_count  = 0;
    s_cursor = -1;

    if (privileged)
        return;  /* never load/save for privileged sessions */

    /* Load from ~/.stsh_history */
    FILE *f = fopen("/root/.stsh_history", "r");
    if (!f) return;

    char line[LINE_SIZE];
    while (fgets(line, LINE_SIZE, f)) {
        int len = (int)strlen(line);
        if (len > 0 && line[len - 1] == '\n')
            line[len - 1] = '\0';
        if (line[0] == '\0') continue;
        strncpy(s_entries[wrap(s_count)], line, LINE_SIZE - 1);
        s_entries[wrap(s_count)][LINE_SIZE - 1] = '\0';
        s_count++;
    }
    fclose(f);
}

void
hist_add(const char *line)
{
    if (!line || line[0] == '\0')
        return;

    /* Suppress duplicate consecutive entries */
    if (s_count > 0 &&
        strcmp(s_entries[wrap(s_count - 1)], line) == 0)
        return;

    strncpy(s_entries[wrap(s_count)], line, LINE_SIZE - 1);
    s_entries[wrap(s_count)][LINE_SIZE - 1] = '\0';
    s_count++;
    s_cursor = -1;
}

const char *
hist_prev(void)
{
    if (s_count == 0) return NULL;

    if (s_cursor == -1)
        s_cursor = s_count - 1;
    else if (s_cursor > 0 && s_cursor > s_count - HIST_SIZE)
        s_cursor--;
    else
        return s_entries[wrap(s_cursor)]; /* at oldest, don't move */

    return s_entries[wrap(s_cursor)];
}

const char *
hist_next(void)
{
    if (s_cursor == -1) return NULL;

    s_cursor++;
    if (s_cursor >= s_count) {
        s_cursor = -1;
        return NULL;  /* back to empty line */
    }
    return s_entries[wrap(s_cursor)];
}

void
hist_reset_cursor(void)
{
    s_cursor = -1;
}

void
hist_save(void)
{
    if (s_privileged)
        return;  /* never persist privileged session history */

    FILE *f = fopen("/root/.stsh_history", "w");
    if (!f) return;

    int start = 0;
    if (s_count > HIST_SIZE)
        start = s_count - HIST_SIZE;

    for (int i = start; i < s_count; i++)
        fprintf(f, "%s\n", s_entries[wrap(i)]);

    fclose(f);
}
```

```c
#include "stsh.h"
#include <stdio.h>
#include <string.h>
#include <unistd.h>
#include <termios.h>

static struct termios s_orig;
static int s_raw = 0;

static void
raw_mode(void)
{
    if (s_raw) return;
    tcgetattr(STDIN_FILENO, &s_orig);
    struct termios raw = s_orig;
    raw.c_lflag &= ~(unsigned)(ICANON | ECHO | ISIG);
    raw.c_cc[VMIN]  = 1;
    raw.c_cc[VTIME] = 0;
    tcsetattr(STDIN_FILENO, TCSANOW, &raw);
    s_raw = 1;
}

static void
cooked_mode(void)
{
    if (!s_raw) return;
    tcsetattr(STDIN_FILENO, TCSANOW, &s_orig);
    s_raw = 0;
}

static void
redraw(const char *prompt, const char *buf, int len, int cursor)
{
    int plen = (int)strlen(prompt);
    write(STDOUT_FILENO, "\r", 1);
    write(STDOUT_FILENO, prompt, (size_t)plen);
    write(STDOUT_FILENO, buf, (size_t)len);

    /* Clear to end of line */
    write(STDOUT_FILENO, "\033[K", 3);

    /* Move cursor to correct position */
    if (cursor < len) {
        char seq[16];
        int n = snprintf(seq, sizeof(seq), "\r\033[%dC", plen + cursor);
        write(STDOUT_FILENO, seq, (size_t)n);
    }
}

static void
insert_char(char *buf, int *len, int *cursor, char ch)
{
    if (*len >= LINE_SIZE - 1) return;
    memmove(buf + *cursor + 1, buf + *cursor, (size_t)(*len - *cursor));
    buf[*cursor] = ch;
    (*len)++;
    (*cursor)++;
    buf[*len] = '\0';
}

static void
delete_back(char *buf, int *len, int *cursor)
{
    if (*cursor <= 0) return;
    memmove(buf + *cursor - 1, buf + *cursor, (size_t)(*len - *cursor));
    (*cursor)--;
    (*len)--;
    buf[*len] = '\0';
}

/*
 * editor_readline — read one line with editing support.
 * Returns the length of the line, or -1 on EOF.
 * buf is NOT newline-terminated.
 */
int
editor_readline(const char *prompt, char *buf, int buflen)
{
    char line[LINE_SIZE];
    int len = 0, cursor = 0;

    line[0] = '\0';
    hist_reset_cursor();

    raw_mode();

    write(STDOUT_FILENO, prompt, strlen(prompt));

    for (;;) {
        unsigned char c;
        int nread = (int)read(STDIN_FILENO, &c, 1);
        if (nread <= 0) {
            cooked_mode();
            return -1;  /* EOF */
        }

        if (c == '\r' || c == '\n') {
            write(STDOUT_FILENO, "\r\n", 2);
            break;
        }

        if (c == 4) {  /* Ctrl-D */
            if (len == 0) {
                cooked_mode();
                return -1;
            }
            continue;
        }

        if (c == 3) {  /* Ctrl-C */
            write(STDOUT_FILENO, "^C\r\n", 4);
            len = 0;
            cursor = 0;
            line[0] = '\0';
            write(STDOUT_FILENO, prompt, strlen(prompt));
            continue;
        }

        if (c == 12) {  /* Ctrl-L — clear screen */
            write(STDOUT_FILENO, "\033[2J\033[H", 7);
            redraw(prompt, line, len, cursor);
            continue;
        }

        if (c == 127 || c == 8) {  /* Backspace */
            delete_back(line, &len, &cursor);
            redraw(prompt, line, len, cursor);
            continue;
        }

        if (c == '\t') {
            char completed[LINE_SIZE];
            if (complete(line, cursor, completed, LINE_SIZE) > 0) {
                strncpy(line, completed, LINE_SIZE - 1);
                line[LINE_SIZE - 1] = '\0';
                len = (int)strlen(line);
                cursor = len;
                redraw(prompt, line, len, cursor);
            }
            continue;
        }

        if (c == 27) {  /* ESC — start of escape sequence */
            unsigned char seq[2];
            if (read(STDIN_FILENO, &seq[0], 1) <= 0) continue;
            if (seq[0] != '[') continue;
            if (read(STDIN_FILENO, &seq[1], 1) <= 0) continue;

            switch (seq[1]) {
            case 'A': {  /* Up */
                const char *prev = hist_prev();
                if (prev) {
                    strncpy(line, prev, LINE_SIZE - 1);
                    line[LINE_SIZE - 1] = '\0';
                    len = (int)strlen(line);
                    cursor = len;
                    redraw(prompt, line, len, cursor);
                }
                break;
            }
            case 'B': {  /* Down */
                const char *next = hist_next();
                if (next) {
                    strncpy(line, next, LINE_SIZE - 1);
                    line[LINE_SIZE - 1] = '\0';
                } else {
                    line[0] = '\0';
                }
                len = (int)strlen(line);
                cursor = len;
                redraw(prompt, line, len, cursor);
                break;
            }
            case 'C':  /* Right */
                if (cursor < len) {
                    cursor++;
                    write(STDOUT_FILENO, "\033[C", 3);
                }
                break;
            case 'D':  /* Left */
                if (cursor > 0) {
                    cursor--;
                    write(STDOUT_FILENO, "\033[D", 3);
                }
                break;
            case 'H':  /* Home */
                cursor = 0;
                redraw(prompt, line, len, cursor);
                break;
            case 'F':  /* End */
                cursor = len;
                redraw(prompt, line, len, cursor);
                break;
            }
            continue;
        }

        /* Printable character */
        if (c >= 32 && c <= 126) {
            insert_char(line, &len, &cursor, (char)c);
            redraw(prompt, line, len, cursor);
        }
    }

    cooked_mode();

    if (len >= buflen) len = buflen - 1;
    memcpy(buf, line, (size_t)len);
    buf[len] = '\0';
    return len;
}
```

```c
#include "stsh.h"
#include <stdio.h>
#include <string.h>
#include <unistd.h>
#include <dirent.h>
#include <sys/stat.h>

#define MAX_CANDIDATES 64

static int
is_first_token(const char *buf, int cursor)
{
    for (int i = 0; i < cursor; i++)
        if (buf[i] == ' ' || buf[i] == '\t')
            return 0;
    return 1;
}

static int
find_word_start(const char *buf, int cursor)
{
    int i = cursor - 1;
    while (i >= 0 && buf[i] != ' ' && buf[i] != '\t')
        i--;
    return i + 1;
}

/*
 * complete — attempt tab completion on the current line.
 * Returns >0 if out was modified, 0 if no completion.
 */
int
complete(const char *buf, int cursor, char *out, int outlen)
{
    int wstart = find_word_start(buf, cursor);
    int wlen   = cursor - wstart;
    if (wlen <= 0) return 0;

    char prefix[128];
    if (wlen > 127) wlen = 127;
    memcpy(prefix, buf + wstart, (size_t)wlen);
    prefix[wlen] = '\0';

    char *candidates[MAX_CANDIDATES];
    int ncand = 0;

    if (is_first_token(buf, cursor)) {
        /* Command completion: scan PATH dirs */
        const char *path = env_get("PATH");
        if (!path) path = "/bin";

        /* Simple: just scan /bin for now (PATH is always /bin) */
        DIR *d = opendir("/bin");
        if (d) {
            struct dirent *ent;
            while ((ent = readdir(d)) && ncand < MAX_CANDIDATES) {
                if (strncmp(ent->d_name, prefix, (size_t)wlen) == 0) {
                    candidates[ncand] = strdup(ent->d_name);
                    if (candidates[ncand]) ncand++;
                }
            }
            closedir(d);
        }
    } else {
        /* File completion: scan directory */
        const char *dir_path = ".";
        const char *file_prefix = prefix;
        char dir_buf[128];

        /* Check if prefix contains a / — split into dir + file parts */
        const char *last_slash = strrchr(prefix, '/');
        if (last_slash) {
            int dlen = (int)(last_slash - prefix);
            if (dlen == 0) {
                dir_path = "/";
            } else {
                if (dlen > 127) dlen = 127;
                memcpy(dir_buf, prefix, (size_t)dlen);
                dir_buf[dlen] = '\0';
                dir_path = dir_buf;
            }
            file_prefix = last_slash + 1;
            wlen = (int)strlen(file_prefix);
        }

        DIR *d = opendir(dir_path);
        if (d) {
            struct dirent *ent;
            while ((ent = readdir(d)) && ncand < MAX_CANDIDATES) {
                if (ent->d_name[0] == '.' && file_prefix[0] != '.')
                    continue;  /* skip hidden files unless prefix starts with . */
                if (strncmp(ent->d_name, file_prefix, (size_t)wlen) == 0) {
                    candidates[ncand] = strdup(ent->d_name);
                    if (candidates[ncand]) ncand++;
                }
            }
            closedir(d);
        }
    }

    if (ncand == 0)
        return 0;

    if (ncand == 1) {
        /* Single match: complete it */
        snprintf(out, (size_t)outlen, "%.*s%s ",
                 wstart, buf, candidates[0]);
        free(candidates[0]);
        return 1;
    }

    /* Multiple matches: find longest common prefix */
    int common = (int)strlen(candidates[0]);
    for (int i = 1; i < ncand; i++) {
        int j = 0;
        while (j < common && candidates[i][j] == candidates[0][j])
            j++;
        common = j;
    }

    /* If common prefix is longer than what we typed, complete to it */
    if (common > wlen) {
        char completion[128];
        memcpy(completion, candidates[0], (size_t)common);
        completion[common] = '\0';
        snprintf(out, (size_t)outlen, "%.*s%s", wstart, buf, completion);

        /* Also display candidates */
        write(STDOUT_FILENO, "\r\n", 2);
        for (int i = 0; i < ncand; i++) {
            write(STDOUT_FILENO, candidates[i], strlen(candidates[i]));
            write(STDOUT_FILENO, "  ", 2);
        }
        write(STDOUT_FILENO, "\r\n", 2);

        for (int i = 0; i < ncand; i++)
            free(candidates[i]);
        return 1;
    }

    /* No further completion possible — just display candidates */
    write(STDOUT_FILENO, "\r\n", 2);
    for (int i = 0; i < ncand; i++) {
        write(STDOUT_FILENO, candidates[i], strlen(candidates[i]));
        write(STDOUT_FILENO, "  ", 2);
    }
    write(STDOUT_FILENO, "\r\n", 2);

    for (int i = 0; i < ncand; i++)
        free(candidates[i]);

    /* Copy buf unchanged so the editor redraws */
    strncpy(out, buf, (size_t)outlen - 1);
    out[outlen - 1] = '\0';
    return 1;  /* signal redraw needed */
}
```

```c
#include "stsh.h"
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>
#include <sys/syscall.h>

#define SYS_CAP_QUERY 362
#define SYS_SPAWN     514

static int s_has_delegate = 0;
static int s_has_query    = 0;

/* Human-readable cap names — index = kind value */
static const char *cap_names[] = {
    "NULL",            /*  0 */
    "VFS_OPEN",        /*  1 */
    "VFS_WRITE",       /*  2 */
    "VFS_READ",        /*  3 */
    "AUTH",            /*  4 */
    "CAP_GRANT",       /*  5 */
    "SETUID",          /*  6 */
    "NET_SOCKET",      /*  7 */
    "NET_ADMIN",       /*  8 */
    "THREAD_CREATE",   /*  9 */
    "PROC_READ",       /* 10 */
    "DISK_ADMIN",      /* 11 */
    "FB",              /* 12 */
    "CAP_DELEGATE",    /* 13 */
    "CAP_QUERY",       /* 14 */
};
#define NUM_CAP_NAMES 15

static const char *
rights_str(uint32_t rights)
{
    static char buf[8];
    int i = 0;
    if (rights & 1) buf[i++] = 'r';
    if (rights & 2) buf[i++] = 'w';
    if (rights & 4) buf[i++] = 'x';
    buf[i] = '\0';
    return buf;
}

static int
cap_name_to_kind(const char *name)
{
    for (int i = 1; i < NUM_CAP_NAMES; i++) {
        if (strcmp(cap_names[i], name) == 0)
            return i;
    }
    return -1;
}

int
has_cap_delegate(void)
{
    return s_has_delegate;
}

/* Called once at startup to cache capability presence */
void
caps_init(void)
{
    cap_slot_t slots[CAP_TABLE_SIZE];
    long ret = syscall(SYS_CAP_QUERY, 0L, (long)slots,
                       (long)sizeof(slots));
    if (ret < 0) return;

    for (int i = 0; i < (int)ret; i++) {
        if (slots[i].kind == 13) s_has_delegate = 1;
        if (slots[i].kind == 14) s_has_query    = 1;
    }
}

void
caps_builtin(int argc, char **argv)
{
    long pid = 0;
    if (argc >= 2)
        pid = atol(argv[1]);

    if (pid != 0 && !s_has_query) {
        fprintf(stderr, "stsh: caps: permission denied (CAP_QUERY not held)\n");
        return;
    }

    cap_slot_t slots[CAP_TABLE_SIZE];
    long ret = syscall(SYS_CAP_QUERY, pid, (long)slots,
                       (long)sizeof(slots));
    if (ret < 0) {
        if (ret == -(long)130)
            fprintf(stderr, "stsh: caps: permission denied (ENOCAP)\n");
        else if (ret == -3)
            fprintf(stderr, "stsh: caps: no such process\n");
        else
            fprintf(stderr, "stsh: caps: error %ld\n", ret);
        return;
    }

    int first = 1;
    for (int i = 0; i < (int)ret; i++) {
        if (slots[i].kind == 0) continue;
        if (!first) write(STDOUT_FILENO, " ", 1);
        first = 0;

        const char *name = (slots[i].kind < NUM_CAP_NAMES)
                           ? cap_names[slots[i].kind]
                           : "UNKNOWN";
        printf("%s(%s)", name, rights_str(slots[i].rights));
    }
    if (!first) write(STDOUT_FILENO, "\n", 1);
}

void
sandbox_builtin(int argc, char **argv, char **envp)
{
    if (!s_has_delegate) {
        fprintf(stderr, "stsh: sandbox: permission denied (CAP_DELEGATE not held)\n");
        return;
    }

    /* Parse: sandbox -allow CAP1,CAP2 -- command args... */
    if (argc < 4 || strcmp(argv[1], "-allow") != 0) {
        fprintf(stderr, "usage: sandbox -allow CAP[,CAP,...] -- command [args...]\n");
        return;
    }

    /* Find -- separator */
    int cmd_start = -1;
    for (int i = 3; i < argc; i++) {
        if (strcmp(argv[i], "--") == 0) {
            cmd_start = i + 1;
            break;
        }
    }
    if (cmd_start < 0 || cmd_start >= argc) {
        fprintf(stderr, "stsh: sandbox: missing -- before command\n");
        return;
    }

    /* Parse allowlist */
    cap_slot_t mask[CAP_TABLE_SIZE];
    memset(mask, 0, sizeof(mask));
    int nmask = 0;

    char allowlist[256];
    strncpy(allowlist, argv[2], 255);
    allowlist[255] = '\0';

    char *tok = allowlist;
    while (*tok && nmask < (int)CAP_TABLE_SIZE) {
        char *comma = strchr(tok, ',');
        if (comma) *comma = '\0';

        int kind = cap_name_to_kind(tok);
        if (kind < 0) {
            fprintf(stderr, "stsh: sandbox: unknown capability '%s'\n", tok);
            return;
        }
        mask[nmask].kind   = (uint32_t)kind;
        mask[nmask].rights = 1 | 2 | 4;  /* all rights */
        nmask++;

        if (comma) tok = comma + 1;
        else break;
    }

    /* Build path */
    char path[256];
    if (argv[cmd_start][0] != '/')
        snprintf(path, sizeof(path), "/bin/%s", argv[cmd_start]);
    else
        snprintf(path, sizeof(path), "%s", argv[cmd_start]);

    /* Build child argv */
    char *child_argv[MAX_ARGV + 1];
    int ci = 0;
    for (int i = cmd_start; i < argc && ci < MAX_ARGV; i++)
        child_argv[ci++] = argv[i];
    child_argv[ci] = NULL;

    /* sys_spawn with cap_mask */
    long pid = syscall(SYS_SPAWN, (long)path, (long)child_argv,
                       (long)envp, (long)-1, (long)mask);
    if (pid < 0) {
        fprintf(stderr, "stsh: sandbox: spawn failed (%ld)\n", pid);
        return;
    }

    int status;
    waitpid((int)pid, &status, 0);
}
```

```c
static long
sys_setfg(long pid)
{
    long ret;
    __asm__ volatile(
        "syscall"
        : "=a"(ret)
        : "0"(360L), "D"(pid)
        : "rcx", "r11", "memory"
    );
    return ret;
}

static void
exec_cmd(cmd_t *cmd, char **envp)
{
    char full[256];
    if (cmd->argv[0][0] != '/') {
        snprintf(full, sizeof(full), "/bin/%s", cmd->argv[0]);
        execve(full, cmd->argv, envp);
    } else {
        execve(cmd->argv[0], cmd->argv, envp);
    }
    fprintf(stderr, "%s: not found\n", cmd->argv[0]);
    _exit(127);
}

/*
 * try_builtin �� check if the command is a builtin.
 * Returns 1 if handled, 0 if not a builtin.
 */
int
try_builtin(cmd_t *cmds, int n, int *last_exit)
{
    if (n != 1 || cmds[0].stdin_file || cmds[0].stderr_to_stdout)
        return 0;

    char *cmd = cmds[0].argv[0];
    if (!cmd) return 0;

    if (strcmp(cmd, "exit") == 0) {
        int code = cmds[0].argv[1] ? atoi(cmds[0].argv[1]) : 0;
        hist_save();
        exit(code);
    }
    if (strcmp(cmd, "cd") == 0) {
        const char *path = cmds[0].argv[1];
        if (!path) path = env_get("HOME");
        if (!path) path = "/";
        if (chdir(path) != 0) {
            perror(path);
            *last_exit = 1;
        } else {
            *last_exit = 0;
        }
        return 1;
    }
    if (strcmp(cmd, "help") == 0) {
        puts("stsh builtins: cd, exit, help, caps, sandbox, export, env");
        *last_exit = 0;
        return 1;
    }
    if (strcmp(cmd, "caps") == 0) {
        int argc = 0;
        while (cmds[0].argv[argc]) argc++;
        caps_builtin(argc, cmds[0].argv);
        *last_exit = 0;
        return 1;
    }
    if (strcmp(cmd, "sandbox") == 0) {
        int argc = 0;
        while (cmds[0].argv[argc]) argc++;
        sandbox_builtin(argc, cmds[0].argv, env_as_array());
        *last_exit = 0;
        return 1;
    }
    if (strcmp(cmd, "export") == 0) {
        if (!cmds[0].argv[1]) {
            env_print_all();
        } else {
            char *eq = strchr(cmds[0].argv[1], '=');
            if (eq) {
                *eq = '\0';
                env_set(cmds[0].argv[1], eq + 1);
            } else {
                fprintf(stderr, "usage: export VAR=value\n");
            }
        }
        *last_exit = 0;
        return 1;
    }
    if (strcmp(cmd, "env") == 0) {
        env_print_all();
        *last_exit = 0;
        return 1;
    }

    return 0;
}

void
run_pipeline(cmd_t *cmds, int n, char **envp, int *last_exit)
{
    int pipes[MAX_PIPELINE - 1][2];
    int i, j;

    for (i = 0; i < n - 1; i++) {
        if (pipe(pipes[i]) < 0) {
            perror("pipe");
            for (j = 0; j < i; j++) {
                close(pipes[j][0]);
                close(pipes[j][1]);
            }
            *last_exit = 1;
            return;
        }
    }

    pid_t pids[MAX_PIPELINE];

    for (i = 0; i < n; i++) {
        pid_t pid = fork();
        if (pid < 0) {
            perror("fork");
            for (j = i; j < n - 1; j++) {
                close(pipes[j][0]);
                close(pipes[j][1]);
            }
            int st;
            for (j = 0; j < i; j++)
                waitpid(pids[j], &st, 0);
            *last_exit = 1;
            return;
        }

        if (pid == 0) {
            if (i > 0)
                dup2(pipes[i-1][0], STDIN_FILENO);
            if (i < n - 1)
                dup2(pipes[i][1], STDOUT_FILENO);

            if (cmds[i].stdin_file) {
                int fd = open(cmds[i].stdin_file, O_RDONLY);
                if (fd < 0) {
                    fprintf(stderr, "%s: cannot open\n", cmds[i].stdin_file);
                    _exit(1);
                }
                dup2(fd, STDIN_FILENO);
                close(fd);
            }

            if (cmds[i].stdout_file && i == n - 1) {
                int fd = open(cmds[i].stdout_file,
                              O_WRONLY | O_CREAT | O_TRUNC, 0644);
                if (fd < 0) {
                    fprintf(stderr, "%s: cannot open for writing\n",
                            cmds[i].stdout_file);
                    _exit(1);
                }
                dup2(fd, STDOUT_FILENO);
                close(fd);
            }

            if (cmds[i].stderr_to_stdout)
                dup2(STDOUT_FILENO, STDERR_FILENO);

            for (j = 0; j < n - 1; j++) {
                close(pipes[j][0]);
                close(pipes[j][1]);
            }

            exec_cmd(&cmds[i], envp);
            _exit(127);
        }

        pids[i] = pid;
    }

    for (j = 0; j < n - 1; j++) {
        close(pipes[j][0]);
        close(pipes[j][1]);
    }

    sys_setfg((long)pids[n - 1]);

    int status = 0;
    for (i = 0; i < n; i++)
        waitpid(pids[i], &status, 0);

    sys_setfg(0);

    /* Extract exit code from last stage */
    if ((status & 0x7f) == 0)
        *last_exit = (status >> 8) & 0xff;
    else
        *last_exit = 128 + (status & 0x7f);
}
```

```c
#include "stsh.h"
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <unistd.h>

static int s_last_exit = 0;

static void
build_prompt(char *prompt, int promptlen)
{
    const char *user = env_get("USER");
    if (!user) user = "aegis";

    char cwd[256];
    if (!getcwd(cwd, sizeof(cwd)))
        strcpy(cwd, "/");

    /* Replace HOME prefix with ~ */
    const char *home = env_get("HOME");
    const char *display_path = cwd;
    char tilde_path[256];
    if (home && home[0] && strncmp(cwd, home, strlen(home)) == 0) {
        snprintf(tilde_path, sizeof(tilde_path), "~%s", cwd + strlen(home));
        display_path = tilde_path;
    }

    char suffix = has_cap_delegate() ? '#' : '$';

    snprintf(prompt, (size_t)promptlen, "%s@aegis:%s%c ",
             user, display_path, suffix);
}

static int
detect_paste(char *buf, int buflen)
{
    unsigned char tmp[4096];
    int n = (int)read(STDIN_FILENO, tmp, sizeof(tmp));
    if (n <= 0) return -1;

    if (n < 4) {
        /* Normal input — put it back via ungetc or just handle directly.
         * Since we're in raw mode, write it into buf for the editor. */
        if (n <= buflen) {
            memcpy(buf, tmp, (size_t)n);
        }
        return n;
    }

    /* Paste detected — count newlines */
    int newlines = 0;
    for (int i = 0; i < n; i++)
        if (tmp[i] == '\n' || tmp[i] == '\r') newlines++;

    printf("\r\nstsh: paste detected (%d line%s). Execute?\r\n",
           newlines + 1, newlines > 0 ? "s" : "");
    write(STDOUT_FILENO, "\xe2\x94\x8c", 3); /* ┌ UTF-8 */
    for (int i = 0; i < 40; i++) write(STDOUT_FILENO, "\xe2\x94\x80", 3); /* ─ */
    write(STDOUT_FILENO, "\r\n", 2);

    /* Display pasted content line by line */
    int start = 0;
    for (int i = 0; i <= n; i++) {
        if (i == n || tmp[i] == '\n' || tmp[i] == '\r') {
            write(STDOUT_FILENO, "\xe2\x94\x82 ", 4); /* │ + space */
            if (i > start)
                write(STDOUT_FILENO, tmp + start, (size_t)(i - start));
            write(STDOUT_FILENO, "\r\n", 2);
            start = i + 1;
            /* Skip \r\n pairs */
            if (i < n - 1 && tmp[i] == '\r' && tmp[i+1] == '\n')
                start = ++i + 1;
        }
    }

    write(STDOUT_FILENO, "\xe2\x94\x94", 3); /* └ */
    for (int i = 0; i < 40; i++) write(STDOUT_FILENO, "\xe2\x94\x80", 3);
    write(STDOUT_FILENO, "\r\n[y/n] ", 8);

    unsigned char ch;
    if (read(STDIN_FILENO, &ch, 1) <= 0) return -1;

    if (ch == 'y' || ch == '\r' || ch == '\n') {
        write(STDOUT_FILENO, "\r\n", 2);
        /* Execute each line */
        start = 0;
        for (int i = 0; i <= n; i++) {
            if (i == n || tmp[i] == '\n' || tmp[i] == '\r') {
                int llen = i - start;
                if (llen > 0 && llen < LINE_SIZE) {
                    char line[LINE_SIZE];
                    memcpy(line, tmp + start, (size_t)llen);
                    line[llen] = '\0';

                    /* Expand variables */
                    char expanded[LINE_SIZE];
                    env_expand(line, expanded, LINE_SIZE, s_last_exit);

                    cmd_t cmds[MAX_PIPELINE];
                    int nc = parse_pipeline(expanded, cmds, MAX_PIPELINE);
                    if (nc > 0) {
                        if (!try_builtin(cmds, nc, &s_last_exit))
                            run_pipeline(cmds, nc, env_as_array(), &s_last_exit);
                    }
                }
                start = i + 1;
                if (i < n - 1 && tmp[i] == '\r' && tmp[i+1] == '\n')
                    start = ++i + 1;
            }
        }
        return 0;  /* handled */
    }

    write(STDOUT_FILENO, "\r\nDiscarded.\r\n", 14);
    return 0;  /* handled (discarded) */
}

int
main(int argc, char **argv, char **envp)
{
    /* Initialize environment from login */
    env_init(envp);

    /* Ensure PATH is set */
    if (!env_get("PATH"))
        env_set("PATH", "/bin");

    /* Initialize capability cache */
    caps_init();

    /* Initialize history — privileged if CAP_DELEGATE held */
    hist_init(has_cap_delegate());

    /* -c mode: execute single command and exit (for vigil/scripts) */
    if (argc >= 3 && strcmp(argv[1], "-c") == 0) {
        char line[LINE_SIZE];
        strncpy(line, argv[2], sizeof(line) - 1);
        line[sizeof(line) - 1] = '\0';

        char expanded[LINE_SIZE];
        env_expand(line, expanded, LINE_SIZE, s_last_exit);

        cmd_t cmds[MAX_PIPELINE];
        int n = parse_pipeline(expanded, cmds, MAX_PIPELINE);
        if (n == 0) return 0;

        /* Handle exec builtin in -c mode */
        if (n == 1 && strcmp(cmds[0].argv[0], "exec") == 0 && cmds[0].argv[1]) {
            char full[256];
            char *prog = cmds[0].argv[1];
            char **exec_argv = &cmds[0].argv[1];
            if (prog[0] != '/') {
                snprintf(full, sizeof(full), "/bin/%s", prog);
                execve(full, exec_argv, env_as_array());
            } else {
                execve(prog, exec_argv, env_as_array());
            }
            perror(prog);
            return 127;
        }

        if (!try_builtin(cmds, n, &s_last_exit))
            run_pipeline(cmds, n, env_as_array(), &s_last_exit);
        return s_last_exit;
    }

    /* Prevent SIGCHLD from terminating shell */
    {
        struct sigaction sa;
        __builtin_memset(&sa, 0, sizeof(sa));
        sa.sa_handler = SIG_IGN;
        sigaction(SIGCHLD, &sa, NULL);
    }

    printf("[SHELL] stsh ready\n");

    /* Interactive REPL */
    for (;;) {
        char prompt[256];
        build_prompt(prompt, sizeof(prompt));

        char line[LINE_SIZE];
        int len = editor_readline(prompt, line, LINE_SIZE);
        if (len < 0) break;  /* EOF */
        if (len == 0) continue;

        /* Add to history */
        hist_add(line);

        /* Expand variables */
        char expanded[LINE_SIZE];
        env_expand(line, expanded, LINE_SIZE, s_last_exit);

        /* Handle ; (sequential commands) */
        char *rest = expanded;
        char *semi;
        while ((semi = strchr(rest, ';')) != NULL) {
            *semi = '\0';
            char segment[LINE_SIZE];
            strncpy(segment, rest, LINE_SIZE - 1);
            segment[LINE_SIZE - 1] = '\0';

            cmd_t cmds[MAX_PIPELINE];
            int n = parse_pipeline(segment, cmds, MAX_PIPELINE);
            if (n > 0) {
                if (!try_builtin(cmds, n, &s_last_exit))
                    run_pipeline(cmds, n, env_as_array(), &s_last_exit);
            }
            rest = semi + 1;
        }

        /* Last (or only) segment */
        cmd_t cmds[MAX_PIPELINE];
        int n = parse_pipeline(rest, cmds, MAX_PIPELINE);
        if (n > 0) {
            if (!try_builtin(cmds, n, &s_last_exit))
                run_pipeline(cmds, n, env_as_array(), &s_last_exit);
        }
    }

    hist_save();
    return s_last_exit;
}
```

For v1, paste detection is deferred to after the basic REPL is working — it can be wired in by changing editor.c's read call to read up to 4096 bytes and checking the count before processing.
